# Move window
:move-window -t destination

# Swap 2 windows
:swap-window -s source -t destination

# Join pane 0 from window 4 to current window
:join-pane -s 4.0

# Join pane 0 from session 1 window 2 to current window
:join-pane -s 1:2.0

# Rotate window
C-o

# Cycle layouts
Space

# Kill window (default is &)
:kill-window

# Send command to all panes in window
:setw synchronize-panes
